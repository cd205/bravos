1. **Create `scripts/run_ingestion.py`** — the daemon entry point:

```python
#!/usr/bin/env python3
"""
Bravos Trading System — Ingestion Daemon Entry Point.

Long-running daemon that polls bravosresearch.com every 5 minutes
for new Trade Alert posts, parses them, and stores signals in PostgreSQL.

Usage:
    python scripts/run_ingestion.py

Process management:
    - SIGTERM / SIGINT: graceful shutdown (quits Chrome driver)
    - schedule library fires BravosScraper.run_cycle() every SCRAPE_INTERVAL_SECONDS
    - Exceptions inside run_cycle are caught by @catch_cycle_exceptions — daemon keeps running

Per decisions: D-01, D-13, D-16.
"""
import signal
import sys
import time
import logging

import schedule

from bravos.config.settings import SCRAPE_INTERVAL_SECONDS
from bravos.ingestion.scraper import BravosScraper

# Configure logging (stdlib per research — not structlog)
logging.basicConfig(
    level=logging.INFO,
    format="%(asctime)s %(levelname)s %(name)s %(message)s",
    datefmt="%Y-%m-%dT%H:%M:%S",
)
logger = logging.getLogger("bravos.ingestion.daemon")

_shutdown = False


def handle_shutdown(signum, frame):
    """Handle SIGTERM/SIGINT for graceful shutdown."""
    global _shutdown
    logger.info("Received signal %s — initiating graceful shutdown", signum)
    _shutdown = True


def main():
    signal.signal(signal.SIGTERM, handle_shutdown)
    signal.signal(signal.SIGINT, handle_shutdown)

    logger.info("Starting ingestion daemon...")
    scraper = BravosScraper()
    scraper.startup()

    # Schedule the scrape cycle
    schedule.every(SCRAPE_INTERVAL_SECONDS).seconds.do(scraper.run_cycle)

    # Run first cycle immediately
    logger.info("Running initial scrape cycle...")
    scraper.run_cycle()

    logger.info("Ingestion daemon started — polling every %ds", SCRAPE_INTERVAL_SECONDS)
    while not _shutdown:
        schedule.run_pending()
        time.sleep(1)

    # Graceful cleanup
    logger.info("Shutting down — closing Chrome driver")
    scraper.shutdown()
    logger.info("Ingestion daemon stopped")


if __name__ == "__main__":
    main()
```

2. **Create `tests/test_ingestion_integration.py`** — integration test against live site:

```python
"""
Integration tests for the full ingestion pipeline.

These tests require:
- Live bravosresearch.com access (with valid credentials in GCP Secret Manager)
- Cloud SQL Auth Proxy running on 127.0.0.1:5432
- Chrome/Chromium installed

Marked with @pytest.mark.integration — skipped by default.
Run explicitly: pytest tests/test_ingestion_integration.py -m integration -x -q

Requirement coverage:
  INGST-01 -> test_login_succeeds
  INGST-02 -> test_category_page_returns_posts
  INGST-03 -> test_dedup_end_to_end
  INGST-06 -> test_signal_stored_with_raw_html
  INGST-07 -> test_session_check_after_login
"""
import os
import pytest

# Custom marker for integration tests
pytestmark = pytest.mark.integration


@pytest.fixture(scope="module")
def live_scraper():
    """Create a BravosScraper with real credentials and Chrome driver.
    Shared across all tests in this module for efficiency (one login).
    """
    from bravos.ingestion.scraper import BravosScraper
    scraper = BravosScraper()
    scraper.startup()
    yield scraper
    scraper.shutdown()


def test_login_succeeds(live_scraper):
    """INGST-01: Scraper logs in with GCP-stored credentials."""
    # After startup(), driver should be on a non-login page
    from bravos.config import settings
    url = live_scraper.driver.current_url
    assert "wp-login" not in url, f"Still on login page: {url}"
    # Verify no login form present
    from selenium.webdriver.common.by import By
    login_fields = live_scraper.driver.find_elements(By.NAME, settings.LOGIN_USERNAME_FIELD)
    assert len(login_fields) == 0, "Login form still present after startup"


def test_session_check_after_login(live_scraper):
    """INGST-07: Session check returns True after successful login."""
    assert live_scraper._check_session() is True


def test_category_page_returns_posts(live_scraper):
    """INGST-02: Category page returns at least 1 post."""
    posts = live_scraper._get_recent_posts(limit=5)
    assert len(posts) >= 1, "No posts found on category page"
    # Verify post structure
    post = posts[0]
    assert "title" in post and post["title"], "Post missing title"
    assert "url" in post and post["url"], "Post missing URL"
    assert "bravosresearch.com" in post["url"], f"Unexpected URL domain: {post['url']}"


def test_signal_stored_with_raw_html(live_scraper, db_connection):
    """INGST-06: Signal stored with non-empty raw_html after processing."""
    posts = live_scraper._get_recent_posts(limit=1)
    assert len(posts) >= 1, "No posts to test"

    # Process the first post (will INSERT or skip if already exists)
    live_scraper._process_post(posts[0])

    # Verify in DB
    with db_connection.cursor() as cur:
        cur.execute(
            "SELECT raw_html, confidence, parse_method, scraped_at FROM signals WHERE post_url = %s",
            (posts[0]["url"],)
        )
        row = cur.fetchone()

    assert row is not None, f"Signal not found in DB for URL: {posts[0]['url']}"
    assert row[0] is not None and len(row[0]) > 0, "raw_html is empty"
    assert row[1] in ("high", "medium", "low"), f"Unexpected confidence: {row[1]}"
    assert row[2] in ("regex", "spacy"), f"Unexpected parse_method: {row[2]}"
    assert row[3] is not None, "scraped_at is NULL"


def test_dedup_end_to_end(live_scraper, db_connection):
    """INGST-03: Processing same post twice does not create duplicate rows."""
    posts = live_scraper._get_recent_posts(limit=1)
    assert len(posts) >= 1

    # Process twice
    live_scraper._process_post(posts[0])
    live_scraper._process_post(posts[0])

    # Verify exactly 1 row
    with db_connection.cursor() as cur:
        cur.execute(
            "SELECT COUNT(*) FROM signals WHERE post_url = %s",
            (posts[0]["url"],)
        )
        count = cur.fetchone()[0]

    assert count == 1, f"Expected 1 row, found {count} — dedup failed"
```
